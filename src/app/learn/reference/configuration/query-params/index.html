SKY UX Builder allows you to track parameters in the URL and expose them through your app via the SkyAppConfig class. Below is an example.

import { Component } from '@angular/core'; import { SkyAppConfig } from '@blackbaud/skyux-builder/runtime'; @Component({ selector: 'my-home', templateUrl: './home.component.html' }) export class HomeComponent { public envId: string; constructor(skyAppConfig: SkyAppConfig) { this.envId = skyAppConfig.params.get('envid'); } } Learn more about the params property in our technical reference.
